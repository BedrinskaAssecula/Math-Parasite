Integration

\[ \large \int_{0}^{p} f(x) \ dx = \lim_{m \to \infty} \sum_{k=1}^{m} \frac{p}{m} \cdot f \left( \frac{pk}{m} \right) \] \[ \large f(x) = a_0 x^0 + a_1 x^1 + a_2 x^2 + \ ... \ + a_{n-2} x^{n_2} + a_{n-1} x^{n-1} + a_{n} x^{n} \] \[ \large \lim_{m \to \infty} \quad \frac{p}{m} \sum_{k=1}^{m} \left( a_0 + a_1 (\frac{pk}{m}) + a_2 (\frac{pk}{m})^2 + \ ... \ + a_{n-2} (\frac{pk}{m})^{n-2} + a_{n-1} (\frac{pk}{m})^{n-1} + a_{n} (\frac{pk}{m})^{n} \right) \] \[ \hspace{-100pt} \large \lim_{m \to \infty} \quad \frac{p}{m} \left( \Bigl( \sum_{k=1}^{m} a_0 \Bigl) + \Bigl( \sum_{k=1}^{m} a_1 ( \frac{pk}{m} ) \Bigl) + \Bigl( \sum_{k=1}^{m} a_2 (\frac{pk}{m})^2 \Bigl) + \ \ ... \ + \Bigl( \sum_{k=1}^{m} a_{n-2} (\frac{pk}{m})^{n-2} \Bigl) + \Bigl( \sum_{k=1}^{m} a_{n-1} (\frac{pk}{m})^{n-1} \Bigl) + \Bigl( \sum_{k=1}^{m} a_{n} (\frac{pk}{m})^{n} \Bigl) \right) \] \[ \hspace{-100pt} \large \lim_{m \to \infty} \quad \frac{p}{m} \left( (a_{0} \cdot \sum_{k=1}^{m} 1) + (a_1 \frac{p}{m} \cdot \sum_{k=1}^{m} k) + (a_2 \frac{p^{2}}{m^{2}} \cdot \sum_{k=1}^{m} k^{2}) + \ ... \ + (a_{n-2} \frac{p^{n-2}}{m^{n-2}} \cdot \sum_{k=1}^{m} k^{n-2}) + (a_{n-1} \frac{p^{n-1}}{m^{n-1}} \cdot \sum_{k=1}^{m} k^{n-1}) + (a_{n} \frac{p^{n}}{m^{n}} \cdot \sum_{k=1}^{m} k^{n}) \right) \] \[ \hspace{-100pt} \large \lim_{m \to \infty} \quad \left( (a_{0} \frac{p}{m} \cdot \sum_{k=1}^{m} 1) + (a_1 \frac{p^{2}}{m^{2}} \cdot \sum_{k=1}^{m} k) + (a_2 \frac{p^{3}}{m^{3}} \cdot \sum_{k=1}^{m} k^{2}) + \ ... \ + (a_{n-2} \frac{p^{n-1}}{m^{n-1}} \cdot \sum_{k=1}^{m} k^{n-2}) + (a_{n-1} \frac{p^{n}}{m^{n}} \cdot \sum_{k=1}^{m} k^{n-1}) + (a_{n} \frac{p^{n+1}}{m^{n+1}} \cdot \sum_{k=1}^{m} k^{n}) \right) \] \[ \hspace{0pt} \large \lim_{m \to \infty} \quad \sum_{j=0}^{n} \bigg( \frac{a_{j} p^{j+1}}{m^{j+1}} \cdot \sum_{k=1}^{n} k^{j} \bigg) \]
